Working with the DOM»

This is called the DOM (Document Object Model).

Make Complex Strings»

In JS, you can make strings like:

(one way)»
let area_code = "415";
let prefix = "555";
let suffix = "1212";
let phone = "(" + area_code + ") "
    + prefix + "-" + suffix;
(another way)»
let area_code = "415";
let prefix = "555";
let suffix = "1212";
let phone =
  `(${area_code}) ${prefix}-${suffix}`;

Note the backticks (`), not regular quotes

function showPets() {

Sorting Pets By Likes»

We want to sort the pets by the number of times they’ve been liked,
putting the most popular pet at the top.

Let’s learn how JS sort works!

Array.sort»

JS can sort strings already:

(console)»
let pets = ["Fluffy", "Dr. Slitherscale", "Spike"];
pets.sort()

However, it doesn’t sort numbers properly:

(console)»
let ages = [1, 3, 2, 11, 25];
ages.sort()

JS sorts by letters, not by numbers.

Sorting Callback»

We can provide JS a callback function to sort numbers:

(console)»
let ages = [1, 3, 2, 11, 25];
ages.sort(function (a, b) { return a - b; })

Sorting by Likes»

pets.js»
function sortByLikes(pets) {
  console.log("sortByLikes", pets);

  pets.sort(function (a, b) {
    return b.numLikes - a.numLikes;
  })
}

When the property name you want is the same as the variable holding the value that you want, you can use JavaScript’s “object shorthand”, like this:

function handleSubmit() {

Finding the Target»

/** Sorts given array by likes, descending. */

function sortByLikes(pets) {
  console.log("sortByLikes", pets);

  pets.sort(function (a, b) {
    return b.numLikes - a.numLikes;
  })
}


/** Loop over sorted pet list and add them all to page. */

function showPets() {
  console.log("showPets");

  sortByLikes(pets);

  let petArea = document.getElementById("pets");
  let html = "";

  for (let i = 0; i < pets.length; i++) {
    let pet = pets[i];
    let title = `<h3>${pet.name} (${pet.type})</h3>`;
    let img = `<img src="${pet.photoUrl}">`;
    let comments = `<p>${pet.comments}</p>`;
    let deleteBtn = `<button id="delete-${i}">Delete</button>`
    let likesBtn = `<button id="like-${i}">Like (${pet.numLikes})</button>`;
    html += `
<div id="pet-${i}" class="pet">
  ${img}
  ${title}
  ${comments}
  ${deleteBtn}
  ${likesBtn}
</div>`;
  }

  petArea.innerHTML = html;
}


/** Handle click on a button: either delete or like and redraw lists */

function handleButton(evt) {
  console.log("handleButton", evt);

  if (evt.target.id.startsWith("delete-")) {
    let petId = Number(evt.target.id.replace("delete-", ""));
    deletePet(petId);
  } else if (evt.target.id.startsWith("like-")) {
    let petId = Number(evt.target.id.replace("like-", ""));
    likePet(petId);
  }
  showPets();
}


/** Delete pet with id. */

function deletePet(id) {
  console.log("deletePet", id);

  pets.splice(id, 1);
}


/** Like pet with id. */

function likePet(id) {
  console.log("likePet", id);

  pets[id].numLikes += 1;
}


/** Handle submit of new pet form and redraw list. */

function handleSubmit(evt) {
  console.log("handleSubmit", evt);

  evt.preventDefault();

  let form = document.getElementById("pet-form");
  let name = form.name.value;
  let type = form.type.value;
  let photoUrl = form.photoUrl.value;
  let comments = form.comments.value;
  addPet(name, type, photoUrl, comments);
  showPets();
}


/** Add pet with given info. */

function addPet(name, type, photoUrl, comments) {
  console.log("addPet", name, type, photoUrl, comments);

  pets.push({
    name: name,
    type: type,
    photoUrl: photoUrl,
    comments: comments,
    numLikes: 0,
  });
}


showPets();
